        \begin{align*}
            &m = \iint \sigma(x,y) \, dA\\[0.25em]
            x_s =&\, \frac{1}{m} \iint x \cdot \sigma(x,y) \, dA\\
            y_s =&\, \frac{1}{m} \iint y \cdot \sigma(x,y) \, dA
        \end{align*}
    \textit{Keine Angabe für $\sigma$: $\sigma = 1$}
