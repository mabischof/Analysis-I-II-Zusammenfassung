\subsubsubsection{Kreis mit Mittelpunkt $(x_o, y_o)$ und Radius $R$} 
\vspace{3pt}
\begin{minipage}{0.5\linewidth}
        \vspace{0.5em}
        \textit{implizit:}
        $$
            (x - x_o)^2  + (y - y_o)^2 = R^2
        $$
        \textit{parametrisiert:}
        \begin{align*}
            x &= x_o + R\cos{t}\\
            y &= y_o + R\sin{t}\\
        \end{align*}
\end{minipage}

\subsubsubsection{Ellipse mit Mittelpunkt $(x_o, y_o)$ und Halba. $a, b$}
\vspace{3pt}
